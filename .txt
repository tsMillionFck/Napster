//-ImportDate--
import { songs } from './data.js'
console.log(songs)

//--Selectors--///
//--container-Selectors--/
const songCover = document.querySelector('#song-cover');
const songTitle = document.querySelector('#song-title');
const songArtist = document.querySelector('#song-artist');
//-Audio-Selectors--//
const songAudio = document.querySelector('#song-audio');
//--ProgressBar-//
const progressBar = document.querySelector('#song-progress-bar')
const progressContainer = document.querySelector('#song-progress-container')
//--button-Selectors--/
const songPrev = document.querySelector('#song-prev');
const songPlayPause = document.querySelector('#song-play-pause');
const songNext = document.querySelector('#song-next');
//--playlist-Selector--//
const playList = document.querySelector('#playlist');
//--playlist-button-Selectors--/
const playListShuffle = document.querySelector('#playlist-shuffle')
const playListRepeat = document.querySelector('#playlist-song-repeat')
const playListLoop = document.querySelector('#playlist-loop');

//--States--//
let currentSongIndex = 1;
let isPlaying = false;
let isRepeat = false;
let isShuffle = false;

//--Functions--//
function loadSong(song) {
    songCover.innerHTML = `<img src="${song.cover}" alt="">`;
    songTitle.textContent = song.name;
    songArtist.textContent = song.artist;
    songAudio.src = song.path;
    console.log(songAudio);
}

function pauseSong() {
        songPlayPause.textContent = 'Play';
        songAudio.pause();
        isPlaying = false;
}
function playSong() {
    const playlistSongActive = document.querySelector('.playlist-each-song.active');
    if (playlistSongActive) {
        playlistSongActive.classList.remove('active');
    }

    const allPlayingSongs = document.querySelectorAll('.playlist-each-song');
    allPlayingSongs[currentSongIndex].classList.add('active')

        songPlayPause.textContent = 'Pause';
        songAudio.play();
        isPlaying = true;
}

function updateProgressBar(e) {
    const {duration, currentTime} = e.target;
    const progressPercentage = currentTime / duration * 100;
    progressBar.style.width = `${progressPercentage}%`;
}
loadSong(songs[currentSongIndex])
function setProgressBar(e) {
    const width = this.clientWidth;
    const clickX = e.offsetX;
    const { duration } = songAudio;
    songAudio.currentTime = (clickX / width) * duration;
}
function renderPlaylist() {
    let playListHtml = '';
    songs.forEach((song, index) => {
        playListHtml += 
        `<li class="playlist-each-song" data-index="${index}">
            <div class="playlist-song">
                <div class="img-cont" style="background-image: url(${song.cover})";></div>
                <div class="song-info">
                    <p class="song-title">${song.name}</p>
                    <p class="song-artist">${song.artist}</p>
                </div>
            </div>
        </li>
                `
    })
    playList.innerHTML = playListHtml;
    console.log("hellp")
}
function nextSong() {
    if(isShuffle) {
        shuffleSong();
    } else {
        currentSongIndex++;
        if (currentSongIndex > (songs.length - 1)){
            currentSongIndex = 0;
            loadSong(songs[currentSongIndex]);
        } else {
            loadSong(songs[currentSongIndex]);
        }
        isPlaying = true;
        playSong()
    }
}
function prevSong() {
    currentSongIndex--;
    if (currentSongIndex < 0) {
        currentSongIndex = songs.length - 1;
        loadSong(songs[currentSongIndex])
    } else{
        loadSong(songs[currentSongIndex])
    }
    isPlaying = true;
    playSong()
}
function handleSongEnd() {
    if (isRepeat) {
        playSong();

    } else {
        nextSong();
    }
}
function shuffleSong() {
    let randomIndex;
    do {
        randomIndex = Math.floor(Math.random() * songs.length)
    } while (randomIndex === currentSongIndex);
    currentSongIndex = randomIndex
    loadSong(songs[currentSongIndex]);
    playSong();
}

renderPlaylist();
//--EventListeners--//
songPlayPause.addEventListener('click', () => {
    if (isPlaying) {
        pauseSong();
    } else {
        playSong();
    }
})
songNext.addEventListener('click', () => {
    nextSong();
})

songPrev.addEventListener('click', () => {
    prevSong();
})
playList.addEventListener('click', (e) => {
    let clickedSong = e.target.closest('.playlist-each-song');
    if (!clickedSong) return;
        const newIndex = parseInt(clickedSong.dataset.index);
        currentSongIndex = newIndex;
        loadSong(songs[currentSongIndex]);
        playSong();
})
playListRepeat.addEventListener('click', () => {
    isRepeat = !isRepeat;
    if (isRepeat) {
        playListRepeat.classList.add('activeButton');
    } else {
        playListRepeat.classList.remove('activeButton');
    }
})
playListShuffle.addEventListener('click', () => {
    isShuffle = !isShuffle;
    if(isShuffle) {
        playListShuffle.classList.add('activeButton')
    } else {
        playListShuffle.classList.remove('activeButton')       
    }
    nextSong()
})
songAudio.addEventListener('ended', handleSongEnd)
songAudio.addEventListener('timeupdate', updateProgressBar)
progressContainer.addEventListener('click', setProgressBar);
